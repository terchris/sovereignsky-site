{{- define "RenderImageSimple" -}} {{- $imgObj := .imgObj -}} {{- $src := .src -}} {{- $alt := .alt -}} {{- end -}} {{- define "RenderImageResponsive" -}} {{/* Responsive Image with webp output for better compression */}} {{- $imgObj := .imgObj -}} {{- $alt := .alt -}} {{- $originalWidth := $imgObj.Width -}} {{- $img800 := $imgObj -}} {{- $img1280 := $imgObj -}} {{- if gt $originalWidth 800 -}} {{- $img800 = $imgObj.Resize "800x webp" -}} {{- end -}} {{- if gt $originalWidth 1280 -}} {{- $img1280 = $imgObj.Resize "1280x webp" -}} {{- end -}} {{- $srcset := printf "%s 800w, %s 1280w" $img800.RelPermalink $img1280.RelPermalink -}} {{- end -}} {{- define "RenderImageCaption" -}} {{- with .caption -}} {{- end -}} {{- end -}} {{- $disableImageOptimizationMD := .Page.Site.Params.disableImageOptimizationMD | default false -}} {{- $urlStr := .Destination | safeURL -}} {{- $url := urls.Parse $urlStr -}} {{- $altText := .Text -}} {{- $caption := .Title -}} {{- $isRemote := findRE "^(https?|data)" $url.Scheme -}} {{- $resource := "" -}} {{- if not $isRemote -}} {{- $resource = or ($.Page.Resources.GetMatch $urlStr) (resources.Get $urlStr) -}} {{- end -}}
